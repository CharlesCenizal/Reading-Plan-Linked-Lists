Book newBook(char* name, int chapters) {
   Book N = malloc(sizeof(bibleBook));
   assert(N!=NULL);
   N->name = name;
   N->chapters = chapters;
   N->next = NULL;
   return(N);
}

  Book head;
  int numItems;
}BibleObj;

typedef struct BibleObj* Bible;

Bible newBible(void){
  Bible B = malloc(sizeof(BibleObj));
  assert(B != NULL);
  B->head = NULL;
  B->numItems = 0;
  return B;


}

// n for name
// c for chapter
void insert(Bible B, char* n,int c){
  if(B->numItems == 0){
    Book Bib = newBook(n,c);
    Bib->next = B->head;
    B->head = Bib;
  }else{
    Book Bib = newBook(n,c);
    Book position = B->head;
    while(position->next != NULL){
      position = position -> next;
    }
    position->next = Bib;
  }
  B->numItems++;
}

void printBible(Bible B){
  Book Bib = B->head;
  while(Bib != NULL){
    printf("Book %s has %d Chapters\n",Bib->name,Bib->chapters);
    Bib = Bib->next;
  }
}
